@Test
    void testActivity() {
        String transactionKey = "123";
        String applicationReferenceNumber = "456";

        // Test Builder and AllArgsConstructor
        Activity activity1 = Activity.builder()
                .transactionKey(transactionKey)
                .applicationReferenceNumber(applicationReferenceNumber)
                .build();

        assertEquals(transactionKey, activity1.getTransactionKey());
        assertEquals(applicationReferenceNumber, activity1.getApplicationReferenceNumber());

        // Test NoArgsConstructor
        Activity activity2 = new Activity();
        assertNotNull(activity2);

        // Test setters
        activity2.setTransactionKey(transactionKey);
        activity2.setApplicationReferenceNumber(applicationReferenceNumber);

        assertEquals(transactionKey, activity2.getTransactionKey());
        assertEquals(applicationReferenceNumber, activity2.getApplicationReferenceNumber());

        // Test equals and hashCode
        assertEquals(activity1, activity2);
        assertEquals(activity1.hashCode(), activity2.hashCode());

        // Test toString
        assertTrue(activity1.toString().contains(transactionKey));
        assertTrue(activity1.toString().contains(applicationReferenceNumber));
    }
