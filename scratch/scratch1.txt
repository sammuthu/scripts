@Test
    void testAction() {
        String actionType = "CREATE";
        String actionDateTime = "2021-01-01T00:00:00Z";

        Action action = Action.builder()
                .actionType(actionType)
                .actionDateTime(actionDateTime)
                .build();

        assertEquals(actionType, action.getActionType());
        assertEquals(actionDateTime, action.getActionDateTime());

        // Test setters
        action.setActionType("UPDATE");
        assertEquals("UPDATE", action.getActionType());

        // Test toString
        assertTrue(action.toString().contains("UPDATE"));

        // Test equals and hashCode
        Action action2 = new Action("UPDATE", actionDateTime);
        assertEquals(action, action2);
        assertEquals(action.hashCode(), action2.hashCode());

        // Test not equals and different hashCode
        Action action3 = new Action("DELETE", actionDateTime);
        assertNotEquals(action, action3);
        assertNotEquals(action.hashCode(), action3.hashCode());
    }
