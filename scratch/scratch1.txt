  // Private constructor to prevent instantiation
    private GsonConfiguration() {
        throw new UnsupportedOperationException("This is a utility class and cannot be instantiated");
    }

    @Test
        public void testPrivateConstructor() throws NoSuchMethodException {
            Constructor<GsonConfiguration> constructor = GsonConfiguration.class.getDeclaredConstructor();
            assertTrue(constructor.canAccess(null));

            constructor.setAccessible(true);

            InvocationTargetException thrown = assertThrows(InvocationTargetException.class, () -> {
                constructor.newInstance();
            });

            assertTrue(thrown.getCause() instanceof UnsupportedOperationException);
            assertEquals("This is a utility class and cannot be instantiated", thrown.getCause().getMessage());
        }
